        def reduce
                a = @n
                b = @d
                
                if a < b
                        x = a
                else
                        x = b
                end

                reducida = false

        while (x>1 && !reducida) do
                if (a%x == 0) && (b%x == 0)
                        a = a/x
                        b = b/x
                        reducida = true
                else
                        x = x - 1
                end
        end

                @n = a
                @d = b
                self.to_s
        end
